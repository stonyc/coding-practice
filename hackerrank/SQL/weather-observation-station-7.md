```sql
SELECT DISTINCT CITY FROM STATION WHERE POSITION('A' IN REVERSE(CITY))=1 \
OR POSITION('E' IN REVERSE(CITY))=1 OR POSITION('I' IN REVERSE(CITY))=1 \
OR POSITION('O' IN REVERSE(CITY))=1 OR POSITION('U' IN REVERSE(CITY))=1;
```
